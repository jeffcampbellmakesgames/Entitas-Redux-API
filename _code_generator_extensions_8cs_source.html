46  public static string[] GetComponentNames(this ComponentData data)
47  {
48  // Attempt to get the actual Type object for this
49  var type = data.GetTypeName().ToType();
50  if (type != null)
51  {
52  var attr = Attribute
53  .GetCustomAttributes(type)
54  .OfType<ComponentNameAttribute>()
55  .SingleOrDefault();
56 
57  if (attr != null)
58  {
59  return attr.componentNames;
60  }
61 
62  // Otherwise if the attribute is inaccessible return the string-calculated type name.
63  return new[]
64  {
65  type.ToCompilableString().ShortTypeName().AddComponentSuffix()
66  };
67  }
68 
69  // Otherwise just return the type name as originally set in the ComponentData instance.
70  return new[]
71  {
72  data.GetTypeName()
73  };
74  }

102  public static string Replace(this string template, ComponentData data, string contextName)
103  {
104  return template
105  .Replace(contextName)
106  .Replace("${ComponentType}", data.GetTypeName())
107  .Replace("${ComponentName}", data.ComponentName())
108  .Replace("${componentName}", data.ComponentName().UppercaseFirst())
109  .Replace("${validComponentName}", data.ComponentNameValidUppercaseFirst())
110  .Replace("${prefixedComponentName}", data.PrefixedComponentName())
111  .Replace("${newMethodParameters}", GetMethodParameters(data.GetMemberData(), true))
112  .Replace("${methodParameters}", GetMethodParameters(data.GetMemberData(), false))
113  .Replace("${newMethodArgs}", GetMethodArgs(data.GetMemberData(), true))
114  .Replace("${methodArgs}", GetMethodArgs(data.GetMemberData(), false))
115  .Replace("${Index}", contextName + LOOKUP + "." + data.ComponentName());
116  }
