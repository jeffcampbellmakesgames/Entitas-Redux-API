149  private void UpdateGroupsComponentAddedOrRemoved(IEntity entity, int index, IComponent component)
150  {
151  var groups = _groupsForIndex[index];
152  if (groups != null)
153  {
154  var events = _groupChangedListPool.Get();
155 
156  var tEntity = (TEntity)entity;
157 
158  for (var i = 0; i < groups.Count; i++)
159  {
160  events.Add(groups[i].HandleEntity(tEntity));
161  }
162 
163  for (var i = 0; i < events.Count; i++)
164  {
165  var groupChangedEvent = events[i];
166  groupChangedEvent?.Invoke(
167  groups[i],
168  tEntity,
169  index,
170  component);
171  }
172 
173  _groupChangedListPool.Push(events);
174  }
175  }

197  private void OnEntityReleased(IEntity entity)

210  private void OnDestroyEntity(IEntity entity)
211  {
212  var tEntity = (TEntity)entity;
213  var removed = _entities.Remove(tEntity);
214  if (!removed)
215  {
216  throw new ContextDoesNotContainEntityException(
217  "'" + this + "' cannot destroy " + tEntity + "!",
218  "This cannot happen!?!");
219  }
220 
221  _entitiesCache = null;
222 
223  OnEntityWillBeDestroyed?.Invoke(this, tEntity);
224 
225  tEntity.InternalDestroy();
226 
227  OnEntityDestroyed?.Invoke(this, tEntity);
228 
229  if (tEntity.RetainCount == 1)
230  {
231  // Can be released immediately without
232  // adding to _retainedEntities
233  tEntity.OnEntityReleased -= _cachedEntityReleased;
234  _reusableEntities.Push(tEntity);
235  tEntity.Release(this);
236  tEntity.RemoveAllOnEntityReleasedHandlers();
237  }
238  else
239  {
240  _retainedEntities.Add(tEntity);
241  tEntity.Release(this);
242  }
243  }

308  public TEntity CreateEntity()
309  {
310  TEntity entity;
311 
312  if (_reusableEntities.Count > 0)
313  {
314  entity = _reusableEntities.Pop();
315  entity.Reactivate(_creationIndex++);
316  }
317  else
318  {
319  entity = _entityFactory();
320  entity.Initialize(
321  _creationIndex++,
322  _totalComponents,
323  _componentPools,
324  _contextInfo,
325  _aercFactory(entity));
326  }
327 
328  _entities.Add(entity);
329  entity.Retain(this);
330  _entitiesCache = null;
331 
332  entity.OnComponentAdded += _cachedEntityChanged;
333  entity.OnComponentRemoved += _cachedEntityChanged;
334  entity.OnComponentReplaced += _cachedComponentReplaced;
335  entity.OnEntityReleased += _cachedEntityReleased;
336  entity.OnDestroyEntity += _cachedDestroyEntity;
337 
338  OnEntityCreated?.Invoke(this, entity);
339 
340  return entity;
341  }

395  public IGroup<TEntity> GetGroup(IMatcher<TEntity> matcher)
396  {
397  IGroup<TEntity> group;
398  if (!_groups.TryGetValue(matcher, out group))
399  {
400  group = new Group<TEntity>(matcher);
401  var entities = GetEntities();
402  for (var i = 0; i < entities.Length; i++)
403  {
404  group.HandleEntitySilently(entities[i]);
405  }
406 
407  _groups.Add(matcher, group);
408 
409  for (var i = 0; i < matcher.Indices.Length; i++)
410  {
411  var index = matcher.Indices[i];
412  if (_groupsForIndex[index] == null)
413  {
414  _groupsForIndex[index] = new List<IGroup<TEntity>>();
415  }
416 
417  _groupsForIndex[index].Add(group);
418  }
419 
420  OnGroupCreated?.Invoke(this, @group);
421  }
422 
423  return group;
424  }
