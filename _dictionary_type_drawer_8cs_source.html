39  public bool HandlesType(Type type)

41  return type.IsGenericType && type.GetGenericTypeDefinition() == typeof(Dictionary<,>);
42  }

44  public object DrawAndGetNewValue(Type memberType, string memberName, object value, object target)
45  {
46  var dictionary = (IDictionary)value;
47  var keyType = memberType.GetGenericArguments()[0];
48  var valueType = memberType.GetGenericArguments()[1];
49  var targetType = target.GetType();
50  if (!_keySearchTexts.ContainsKey(targetType))
51  {
52  _keySearchTexts.Add(targetType, string.Empty);
53  }
54 
55  EditorGUILayout.BeginHorizontal();
56  {
57  if (dictionary.Count == 0)
58  {
59  EditorGUILayout.LabelField(memberName, "empty");
60  _keySearchTexts[targetType] = string.Empty;
61  }
62  else
63  {
64  EditorGUILayout.LabelField(memberName);
65  }
66 
67  var keyTypeName = keyType.ToCompilableString().ShortTypeName();
68  var valueTypeName = valueType.ToCompilableString().ShortTypeName();
69  if (EditorGUILayoutTools.MiniButton("new <" + keyTypeName + ", " + valueTypeName + ">"))
70  {
71  if (EntityDrawer.CreateDefault(keyType, out var defaultKey))
72  {
73  if (EntityDrawer.CreateDefault(valueType, out var defaultValue))
74  {
75  dictionary[defaultKey] = defaultValue;
76  }
77  }
78  }
79  }
80  EditorGUILayout.EndHorizontal();
81 
82  if (dictionary.Count > 0)
83  {
84  var indent = EditorGUI.indentLevel;
85  EditorGUI.indentLevel = indent + 1;
86 
87  if (dictionary.Count > 5)
88  {
89  EditorGUILayout.Space();
90  _keySearchTexts[targetType] = EditorGUILayoutTools.SearchTextField(_keySearchTexts[targetType]);
91  }
92 
93  EditorGUILayout.Space();
94 
95  var keys = new ArrayList(dictionary.Keys);
96  for (var i = 0; i < keys.Count; i++)
97  {
98  var key = keys[i];
99  if (EditorGUILayoutTools.MatchesSearchString(key.ToString().ToLower(), _keySearchTexts[targetType].ToLower()))
100  {
101  EntityDrawer.DrawObjectMember(
102  keyType,
103  "key",
104  key,
105  target,
106  (newComponent, newValue) =>
107  {
108  var tmpValue = dictionary[key];
109  dictionary.Remove(key);
110  if (newValue != null)
111  {
112  dictionary[newValue] = tmpValue;
113  }
114  });
115 
116  EntityDrawer.DrawObjectMember(
117  valueType,
118  "value",
119  dictionary[key],
120  target,
121  (newComponent, newValue) => dictionary[key] = newValue);
122 
123  EditorGUILayout.Space();
124  }
125  }
126 
127  EditorGUI.indentLevel = indent;
128  }
129 
130  return dictionary;
131  }
132  }
