288  public override Systems Add(ISystem system)
289  {
290  _systems.Add(system);
291 
292  SystemInfo childSystemInfo;
293 
294  if (system is DebugSystems debugSystems)
295  {
296  childSystemInfo = debugSystems.SystemInfo;
297  debugSystems.GameObject.transform.SetParent(_gameObject.transform, false);
298  }
299  else
300  {
301  childSystemInfo = new SystemInfo(system);
302  }
303 
304  _allSystemInfos.Add(childSystemInfo);
305 
306  childSystemInfo.parentSystemInfo = _systemInfo;
307 
308  if (childSystemInfo.IsInitializeSystems)
309  {
310  _initializeSystemInfos.Add(childSystemInfo);
311  }
312 
313  if (childSystemInfo.IsUpdateSystems || childSystemInfo.IsReactiveSystems)
314  {
315  _updateSystemInfos.Add(childSystemInfo);
316  }
317 
318  if (childSystemInfo.IsFixedUpdateSystems)
319  {
320  _fixedUpdateSystemInfos.Add(childSystemInfo);
321  }
322 
323  if (childSystemInfo.IsLateUpdateSystems)

380  public override void Update()
381  {
382  _updateDuration = 0;
383  if (Time.frameCount % (int)avgResetInterval == 0)
384  {
385  ResetDurations();
386  }
387

392  {
393  _stopwatch.Reset();
394  _stopwatch.Start();
395  _updateSystems[i].Update();
396  _stopwatch.Stop();
397  var duration = _stopwatch.Elapsed.TotalMilliseconds;
398  _updateDuration += duration;

400  }
