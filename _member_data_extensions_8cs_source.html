92  public static string GetMemberCopyAssignment(this MemberData memberData, Type memberType)
93  {
94  var result = string.Empty;
95 
96  // In this case, a mutable reference type
97  var isUnityObjectType = memberType.IsAssignableFrom(typeof(UnityEngine.Object));
98  if (memberType.IsMutableReferenceType() && !isUnityObjectType)
99  {
100  var hasDefaultConstructor = memberType.HasDefaultConstructor();
101  if (memberType.IsList(out var genericType) && hasDefaultConstructor)
102  {
103  result = GetListMemberCopyAssignment(memberData, genericType);
104  }
105  else if (memberType.IsDictionary(out var genericKeyType, out var genericValueType) &&
106  hasDefaultConstructor)
107  {
108  result = GetDictionaryMemberCopyAssignment(memberData, genericKeyType, genericValueType);
109  }
110  else if (memberType.IsArray)
111  {
112  var elementType = memberType.GetElementType();
113  result = GetArrayMemberCopyAssignment(memberData, elementType);
114  }
115  // Otherwise if the member type implements ICloneable, clone the instance as assignment
116  else if (memberType.ImplementsInterface<ICloneable>())
117  {
118  result = MEMBER_CLONE_ASSIGNMENT.Replace(memberData);
119  }
120  }
121 
122  return result;
123  }

146  public static string GetDictionaryMemberCopyAssignment(
147  MemberData memberData,
148  Type genericKeyType,
149  Type genericValueType)
150  {
151  var result = string.Empty;
152  // Check for more specific deep copy implementation potential
153  if (genericValueType.IsList(out var genericListType) &&
154  genericListType.IsMutableReferenceType() &&
155  genericListType.ImplementsInterface<ICloneable>())
156  {
157  result = DICTIONARY_MEMBER_LIST_DEEP_COPY_ASSIGNMENT.Replace(memberData);
158  }
159  else if (genericValueType.IsArray)
160  {
161  var elementType = genericValueType.GetElementType();
162  if (elementType.IsMutableReferenceType() && elementType.ImplementsInterface<ICloneable>())
163  {
164  result = DICTIONARY_MEMBER_ARRAY_DEEP_COPY_ASSIGNMENT.Replace(memberData);
165  }
166  }
167 
168  // If we can't find a more specific deep copy...
169  if (string.IsNullOrEmpty(result))
170  {
171  // ...and the value type is capable of deep copy, use a 1-level deep copy,...
172  if (genericValueType.IsMutableReferenceType() &&
173  genericValueType.ImplementsInterface<ICloneable>())
174  {
175  result = DICTIONARY_MEMBER_DEEP_COPY_ASSIGNMENT.Replace(memberData);
176  }
177  // ...otherwise if the value ends up not being capable of a deep copy, use shallow copy instead
178  else
179  {
180  result = DICTIONARY_MEMBER_SHALLOW_COPY_ASSIGNMENT.Replace(memberData);
181  }
182  }
183 
184  return result;
185  }
