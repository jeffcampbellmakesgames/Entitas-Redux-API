183  {
184  var contextInfo = entity.ContextInfo;
185  var infosList = new List<ComponentInfo>(contextInfo.componentTypes.Length);
186  for (var i = 0; i < contextInfo.componentTypes.Length; i++)
187  {

189  new ComponentInfo

191  index = i, name = contextInfo.componentNames[i], type = contextInfo.componentTypes[i]
192  });
193  }
194 
195  infos = infosList.ToArray();
196  ContextToComponentInfos.Add(entity.ContextInfo.name, infos);
197  }
198
