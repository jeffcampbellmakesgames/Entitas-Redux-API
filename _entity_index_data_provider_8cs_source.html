53  private EntityIndexData[] CreateEntityIndexData(Type type, List<PublicMemberInfo> infos)
54  {
55  var hasMultiple = infos.Count(i => i.attributes.Count(attr => attr.attribute is AbstractEntityIndexAttribute) == 1) >
56  1;
57  return infos
58  .Where(i => i.attributes.Count(attr => attr.attribute is AbstractEntityIndexAttribute) == 1)
59  .Select(
60  info =>
61  {
62  var data = new EntityIndexData();
63  var attribute = (AbstractEntityIndexAttribute)info.attributes
64  .Single(attr => attr.attribute is AbstractEntityIndexAttribute)
65  .attribute;
66 
67  data.SetEntityIndexType(GetEntityIndexType(attribute));
68  data.IsCustom(false);
69  data.SetEntityIndexName(type.ToCompilableString().ToComponentName());
70  data.SetKeyType(info.type.ToCompilableString());
71  data.SetComponentType(type.ToCompilableString());
72  data.SetMemberName(info.name);
73  data.SetHasMultiple(hasMultiple);
74  data.SetContextNames(_contextsComponentDataProvider.GetContextNamesOrDefault(type));
75 
76  return data;
77  })
78  .ToArray();
79  }

81  private EntityIndexData CreateCustomEntityIndexData(Type type)
82  {
83  var data = new EntityIndexData();
84 
85  var attribute = (CustomEntityIndexAttribute)type.GetCustomAttributes(typeof(CustomEntityIndexAttribute), false)[0];
86 
87  data.SetEntityIndexType(type.ToCompilableString());
88  data.IsCustom(true);
89  data.SetEntityIndexName(type.ToCompilableString().RemoveDots());
90  data.SetHasMultiple(false);
91  data.SetContextNames(
92  new[]
93  {
94  attribute.contextType.ToCompilableString().ShortTypeName().RemoveContextSuffix()
95  });
96 
97  var getMethods = type
98  .GetMethods(BindingFlags.Public | BindingFlags.Instance)
99  .Where(method => Attribute.IsDefined(method, typeof(EntityIndexGetMethodAttribute)))
100  .Select(
101  method => new MethodData(
102  method.ReturnType.ToCompilableString(),
103  method.Name,
104  method.GetParameters()
105  .Select(p => new MemberData(p.ParameterType, p.Name))
106  .ToArray()))
107  .ToArray();
108 
109  data.SetCustomMethods(getMethods);
110 
111  return data;
112  }

143  public CodeGeneratorData[] GetData()
144  {
145  var assemblies = _assembliesConfig.DoUseWhitelistOfAssemblies
146  ? ReflectionTools.GetAvailableAssemblies(_assembliesConfig.WhiteListedAssemblies)
147  : ReflectionTools.GetAvailableAssemblies();
148 
149  var types = _types ??
150  assemblies.SelectMany(x => x.GetTypes());
151 
152  var entityIndexData = types
153  .Where(type => !type.IsAbstract)
154  .Where(type => type.ImplementsInterface<IComponent>())
155  .ToDictionary(
156  type => type,
157  type => type.GetPublicMemberInfos())
158  .Where(kv => kv.Value.Any(info => info.attributes.Any(attr => attr.attribute is AbstractEntityIndexAttribute)))
159  .SelectMany(kv => CreateEntityIndexData(kv.Key, kv.Value));
160 
161  var customEntityIndexData = types
162  .Where(type => !type.IsAbstract)
163  .Where(type => Attribute.IsDefined(type, typeof(CustomEntityIndexAttribute)))
164  .Select(CreateCustomEntityIndexData);
165 
166  return entityIndexData
167  .Concat(customEntityIndexData)
168  .ToArray();
169  }
