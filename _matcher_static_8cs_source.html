88  private static int[] MergeIndices(IMatcher<TEntity>[] matchers)
89  {
90  var indices = new int[matchers.Length];
91  for (var i = 0; i < matchers.Length; i++)
92  {

94  if (matcher.Indices.Length != 1)
95  {
96  throw new MatcherException(matcher.Indices.Length);
97  }
98 
99  indices[i] = matcher.Indices[0];
100  }

102  return indices;
