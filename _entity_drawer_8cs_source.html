359  public static void DrawComponent(bool[] unfoldedComponents,
360  string[] componentMemberSearch,
361  IEntity entity,
362  int index,
363  IComponent component)
364  {
365  var componentType = component.GetType();
366  var componentName = componentType.Name.RemoveComponentSuffix();
367  if (EditorGUILayoutTools.MatchesSearchString(componentName.ToLower(), ComponentNameSearchString.ToLower()))
368  {
369  var boxStyle = GetColoredBoxStyle(entity, index);
370 
371  EditorGUILayout.BeginVertical(boxStyle);
372 
373  if (!Attribute.IsDefined(componentType, typeof(DontDrawComponentAttribute)))
374  {
375  var memberInfos = componentType.GetPublicMemberInfos();
376  EditorGUILayout.BeginHorizontal();
377  {
378  if (memberInfos.Count == 0)
379  {
380  EditorGUILayout.LabelField(componentName, EditorStyles.boldLabel);
381  }
382  else
383  {
384  unfoldedComponents[index] = EditorGUILayoutTools.Foldout(
385  unfoldedComponents[index],
386  componentName,
387  FoldoutStyle);
388  if (unfoldedComponents[index])
389  {
390  componentMemberSearch[index] = memberInfos.Count > 5
391  ? EditorGUILayoutTools.SearchTextField(componentMemberSearch[index])
392  : string.Empty;
393  }
394  }
395 
396  if (EditorGUILayoutTools.MiniButton("-"))
397  {
398  entity.RemoveComponent(index);
399  }
400  }
401  EditorGUILayout.EndHorizontal();
402 
403  if (unfoldedComponents[index])
404  {
405  var newComponent = entity.CreateComponent(index, componentType);
406  component.CopyPublicMemberValues(newComponent);
407 
408  var changed = false;
409  var componentDrawer = GetComponentDrawer(componentType);
410  if (componentDrawer != null)
411  {
412  EditorGUI.BeginChangeCheck();
413  {
414  componentDrawer.DrawComponent(newComponent);
415  }
416  changed = EditorGUI.EndChangeCheck();
417  }
418  else
419  {
420  foreach (var info in memberInfos)
421  {
422  if (EditorGUILayoutTools.MatchesSearchString(
423  info.name.ToLower(),
424  componentMemberSearch[index].ToLower()))
425  {
426  var memberValue = info.GetValue(newComponent);
427  var memberType = memberValue == null ? info.type : memberValue.GetType();
428  if (DrawObjectMember(
429  memberType,
430  info.name,
431  memberValue,
432  newComponent,
433  info.SetValue))
434  {
435  changed = true;
436  }
437  }
438  }
439  }
440 
441  if (changed)
442  {
443  entity.ReplaceComponent(index, newComponent);
444  }
445  else
446  {
447  entity.GetComponentPool(index).Push(newComponent);
448  }
449  }
450  }
451  else
452  {
453  EditorGUILayout.LabelField(componentName, "[DontDrawComponent]", EditorStyles.boldLabel);
454  }
455 
456  EditorGUILayoutTools.EndVerticalBox();
457  }
458  }

545  public static bool DrawObjectMember(Type memberType, string memberName, object value, object target,
546  Action<object, object> setValue)
547  {
548  if (value == null)
549  {
550  EditorGUI.BeginChangeCheck();
551  {
552  var isUnityObject = memberType == typeof(Object) || memberType.IsSubclassOf(typeof(Object));
553  EditorGUILayout.BeginHorizontal();
554  {
555  if (isUnityObject)
556  {
557  setValue(
558  target,
559  EditorGUILayout.ObjectField(
560  memberName,
561  (Object)value,
562  memberType,
563  true));
564  }
565  else
566  {
567  EditorGUILayout.LabelField(memberName, "null");
568  }
569 
570  if (EditorGUILayoutTools.MiniButton("new " + memberType.ToCompilableString().ShortTypeName()))
571  {
572  if (CreateDefault(memberType, out var defaultValue))
573  {
574  setValue(target, defaultValue);
575  }
576  }
577  }
578  EditorGUILayout.EndHorizontal();
579  }
580 
581  return EditorGUI.EndChangeCheck();
582  }
583 
584  if (!memberType.IsValueType)
585  {
586  EditorGUILayout.BeginHorizontal();
587  EditorGUILayout.BeginVertical();
588  }
589 
590  EditorGUI.BeginChangeCheck();
591  {
592  var typeDrawer = GetTypeDrawer(memberType);
593  if (typeDrawer != null)
594  {
595  var newValue = typeDrawer.DrawAndGetNewValue(
596  memberType,
597  memberName,
598  value,
599  target);
600  setValue(target, newValue);
601  }
602  else
603  {
604  var targetType = target.GetType();
605  var shouldDraw = !targetType.ImplementsInterface<IComponent>() ||
606  !Attribute.IsDefined(targetType, typeof(DontDrawComponentAttribute));
607  if (shouldDraw)
608  {
609  EditorGUILayout.LabelField(memberName, value.ToString());
610 
611  var indent = EditorGUI.indentLevel;
612  EditorGUI.indentLevel += 1;
613 
614  EditorGUILayout.BeginVertical();
615  {
616  foreach (var info in memberType.GetPublicMemberInfos())
617  {
618  var mValue = info.GetValue(value);
619  var mType = mValue == null ? info.type : mValue.GetType();
620  DrawObjectMember(
621  mType,
622  info.name,
623  mValue,
624  value,
625  info.SetValue);
626  if (memberType.IsValueType)
627  {
628  setValue(target, value);
629  }
630  }
631  }
632  EditorGUILayout.EndVertical();
633 
634  EditorGUI.indentLevel = indent;
635  }
636  else
637  {
638  DrawUnsupportedType(memberType, memberName, value);
639  }
640  }
641 
642  if (!memberType.IsValueType)
643  {
644  EditorGUILayout.EndVertical();
645  if (EditorGUILayoutTools.MiniButton(""))
646  {
647  setValue(target, null);
648  }
649 
650  EditorGUILayout.EndHorizontal();
651  }
652  }
653 
654  return EditorGUI.EndChangeCheck();
655  }

657  public static bool CreateDefault(Type type, out object defaultValue)
658  {
659  try
660  {
661  defaultValue = Activator.CreateInstance(type);
662  return true;
663  }
664  catch (Exception)
665  {
666  foreach (var creator in DEFAULT_INSTANCE_CREATORS)
667  {
668  if (creator.HandlesType(type))
669  {
670  defaultValue = creator.CreateDefault(type);
671  return true;
672  }
673  }
674  }
675 
676  defaultValue = null;
677  return false;
678  }
