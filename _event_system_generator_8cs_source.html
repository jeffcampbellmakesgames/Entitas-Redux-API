124  private CodeGenFile[] Generate(string contextName, ComponentData data)
125  {
126  return data.GetEventData()
127  .Select(
128  eventData =>
129  {
130  var methodArgs = data.GetEventMethodArgs(
131  eventData,
132  ", " +
133  (data.GetMemberData().Length == 0
134  ? data.PrefixedComponentName()
135  : GetMethodArgs(data.GetMemberData())));
136 
137  var cachedAccess = data.GetMemberData().Length == 0
138  ? string.Empty
139  : "var component = e." + data.ComponentNameValidUppercaseFirst() + ";";
140 
141  if (eventData.eventType == EventType.Removed)
142  {
143  methodArgs = string.Empty;
144  cachedAccess = string.Empty;
145  }
146 
147  var template = eventData.eventTarget == EventTarget.Self
148  ? SELF_TARGET_TEMPLATE
149  : ANY_TARGET_TEMPLATE;
150 
151  var fileContent = template
152  .Replace("${GroupEvent}", eventData.eventType.ToString())
153  .Replace("${filter}", GetFilter(data, contextName, eventData))
154  .Replace("${cachedAccess}", cachedAccess)
155  .Replace("${methodArgs}", methodArgs)
156  .Replace(data, contextName, eventData);
157 
158  return new CodeGenFile(
159  "Events" +
160  Path.DirectorySeparatorChar +
161  "Systems" +
162  Path.DirectorySeparatorChar +
163  data.Event(contextName, eventData) +
164  "EventSystem.cs",
165  fileContent,
166  GetType().FullName);
167  })
168  .ToArray();
169  }

171  private string GetFilter(ComponentData data, string contextName, EventData eventData)
172  {
173  var filter = string.Empty;
174  if (data.GetMemberData().Length == 0)
175  {
176  switch (eventData.eventType)
177  {
178  case EventType.Added:
179  filter = "entity." + data.PrefixedComponentName();
180  break;
181  case EventType.Removed:
182  filter = "!entity." + data.PrefixedComponentName();
183  break;
184  }
185  }
186  else
187  {
188  switch (eventData.eventType)
189  {
190  case EventType.Added:
191  filter = "entity.has" + data.ComponentName();
192  break;
193  case EventType.Removed:
194  filter = "!entity.has" + data.ComponentName();
195  break;
196  }
197  }
198 
199  if (eventData.eventTarget == EventTarget.Self)
200  {
201  filter += " && entity.has" + data.EventListener(contextName, eventData);
202  }
203 
204  return filter;
205  }
